1. ##### 解数独

   输入一个`9×9`的二维数组`board`，其中有些位置已经填了数字，没填的使用`"."`占位

   将`board`填满并且满足以下三个要求：

   - 数字 `1-9` 在每一行只能出现一次。
   - 数字 `1-9` 在每一列只能出现一次。
   - 数字 `1-9` 在每一个以粗实线分隔的 `3x3` 宫内只能出现一次。（请参考示例图）

* 和`N`皇后问题很像，不过每一行需要填入的九个数字，并且保证不重复，故而回溯过程中需要经过每一个格子，并且每个格子都要填入数字，填入的过程中进行相应剪枝。以下就是每填入一个数字之前都判断是否合理

  ```js
  /**
   * @param {character[][]} board
   * @return {void} Do not return anything, modify board in-place instead.
   */
  var solveSudoku = function (board) {
      /* 回溯，每填一个位置都判断是否合理，进行剪枝 */
  
      /* 传入当前要填的行以及列 */
      const backTracking = function (row, col) {
          if (row === 8 && col === 9) {
              /* 如果最后一行已经填完了 */
              return true;
          } else if (col === 9) {
              /* 如果当前行填完了，填下一行 */
              if (backTracking(row + 1, 0)) {
                  return true;
              }
          } else if (board[row][col] !== ".") {
              /* 如果当前格子不是空白 */
              if (backTracking(row, col + 1)) {
                  return true;
              }
          } else {
              /* 开始填 */
              for (let i = 1; i <= 9; i++) {
                  /* 首先判断是否用过，然后再回溯 */
                  if (isValid(i, row, col)) {
                      board[row][col] = i.toString();
                      if (backTracking(row, col + 1)) {
                          return true;
                      }
                      board[row][col] = "."; /* 回溯 */
                  }
              }
          }
          return false;
      };
  
      /* 判断当前数字是否合格 */
      function isValid(num, row, col) {
          /* 如果当前行用过了 */
          for (let i = 0; i < 9; i++) {
              if (board[row][i] === num.toString()) {
                  return false;
              }
          }
  
          /* 如果当前列用过了 */
          for (let i = 0; i < 9; i++) {
              if (board[i][col] === num.toString()) {
                  return false;
              }
          }
  
          /* 如果所在九宫格用过了 */
          let i = row - (row % 3);
          let j = col - (col % 3);
          let maxI = i + 3;
          let maxJ = j + 3;
          for (; i < maxI; i++) {
              j = col - (col % 3);
              for (; j < maxJ; j++) {
                  if (board[i][j] === num.toString()) {
                      return false;
                  }
              }
          }
  
          return true;
      }
  
      backTracking(0, 0);
  };
  ```

* 由于上面每填一个数字都要遍历一遍行，列，所在三宫格判断是否使用过，时间复杂度过高，可以牺牲空间换时间，使用三个`9×9`的二维数组记录每个维度数字的使用情况，首先遍历一个`board`，将本来已经使用过的数字记录一下

  ```js
  /**
   * @param {character[][]} board
   * @return {void} Do not return anything, modify board in-place instead.
   */
  var solveSudoku = function (board) {
      /* 遍历一遍数组，记录每一行的填入情况,每一行的大小为9，记录每个数字在当前行是否填过了 */
      const rowUsed = [];
      /* 同理，记录列和每个3×3宫的填写情况 */
      const colUsed = [];
      const threeGaleUsed = [];
      for (let i = 0; i < 9; i++) {
          rowUsed[i] = [];
          colUsed[i] = [];
          threeGaleUsed[i] = [];
      }
      /* 遍历一遍，初始化填入的情况 */
      for (let i = 0; i < 9; i++) {
          for (let j = 0; j < 9; j++) {
              let num = board[i][j];
              if (num !== ".") {
                  num = parseInt(num) - 1;
                  /* 如果不为空 */
                  rowUsed[i][num] = true;
                  colUsed[j][num] = true;
                  /* 首先得到当前所在三宫格左上角坐标 */
                  let startX = i - i % 3;
                  let startY = j - j % 3;
                  threeGaleUsed[startX + startY / 3][num] = true;
              }
          }
      }
  
      const backTracking = function (row, col) {
          /* 回溯，传入当前要填的格子坐标 */
          /* 如果已经整个填完 */
          if (row === 8 && col === 9) {
              return true;
          } else if (col === 9) {
              /* 如果本行已经填完 */
              if (backTracking(row + 1, 0)) {
                  return true;
              }
          } else if (board[row][col] !== ".") {
              /* 如果不为空，填下一个 */
              if (backTracking(row, col + 1)) {
                  return true;
              }
          } else {
              /* 否则就开始填 */
              for (let i = 1; i <= 9; i++) {
                  let num = i - 1;/* 映射到[0,8]区间 */
                  /* 开始判断是否填过 */
                  if (rowUsed[row][num] === true) {
                      continue;
                  }
                  if (colUsed[col][num] === true) {
                      continue;
                  }
                  let startX = row - row % 3;
                  let startY = col - col % 3;
                  let threeGale = startX + startY / 3;
                  if (threeGaleUsed[threeGale][num] === true) {
                      continue;
                  }
                  /* 否则填入 */
                  board[row][col] = i.toString();
                  rowUsed[row][num] = true;
                  colUsed[col][num] = true;
                  threeGaleUsed[threeGale][num] = true;
                  if (backTracking(row, col + 1) === true) {
                      return true;
                  }
                  board[row][col] = ".";
                  rowUsed[row][num] = false;
                  colUsed[col][num] = false;
                  threeGaleUsed[threeGale][num] = false;
              }
          }
          return false;
      }
  
      backTracking(0, 0);
  };
  ```
